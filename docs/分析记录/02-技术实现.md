```typescript
// src/utils/storage/opportunityIndexedDB.ts

// 添加记录
export async function addOpportunityRecord(record: OpportunityRecord): Promise<void> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onsuccess = (event) => {
      const db = (event.target as IDBOpenDBRequest).result;
      const transaction = db.transaction(['opportunity_records'], 'readwrite');
      const store = transaction.objectStore('opportunity_records');

      store.add(record);

      transaction.oncomplete = () => resolve();
      transaction.onerror = () => reject(transaction.error);
    };
  });
}

// 获取所有记录
export async function getAllOpportunityRecords(): Promise<OpportunityRecord[]> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onsuccess = (event) => {
      const db = (event.target as IDBOpenDBRequest).result;
      const transaction = db.transaction(['opportunity_records'], 'readonly');
      const store = transaction.objectStore('opportunity_records');

      const getAllRequest = store.getAll();
      getAllRequest.onsuccess = () => resolve(getAllRequest.result);
      getAllRequest.onerror = () => reject(getAllRequest.error);
    };
  });
}

// 按日期范围查询
export async function getOpportunityRecordsByDateRange(
  startDate: string,
  endDate: string
): Promise<OpportunityRecord[]> {
  const allRecords = await getAllOpportunityRecords();

  return allRecords.filter((record) => {
    return record.date >= startDate && record.date <= endDate;
  });
}

// 清空所有记录
export async function clearAllOpportunityRecords(): Promise<void> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onsuccess = (event) => {
      const db = (event.target as IDBOpenDBRequest).result;
      const transaction = db.transaction(['opportunity_records'], 'readwrite');
      const store = transaction.objectStore('opportunity_records');

      store.clear();

      transaction.oncomplete = () => resolve();
      transaction.onerror = () => reject(transaction.error);
    };
  });
}
```

## ⚙️ 统计服务

### 股票统计计算

**文件**: `src/services/opportunity/recordService.ts`

```typescript
import {
  getAllOpportunityRecords,
  getOpportunityRecordsByDateRange,
} from '@/utils/storage/opportunityIndexedDB';
import type { StockStatistics } from '@/types/stock';

/**
 * 计算股票统计数据
 * @param dateRange 可选的日期范围
 * @returns 股票统计数组
 */
export async function calculateStockStatistics(dateRange?: {
  startDate: string;
  endDate: string;
}): Promise<StockStatistics[]> {
  // 1. 获取记录
  const records = dateRange
    ? await getOpportunityRecordsByDateRange(dateRange.startDate, dateRange.endDate)
    : await getAllOpportunityRecords();

  // 2. 按股票代码分组
  const stockMap = new Map<string, StockStatistics>();

  records.forEach((record) => {
    const { code, name, date, concepts, industry } = record;

    if (!stockMap.has(code)) {
      stockMap.set(code, {
        code,
        name,
        count: 0,
        latestDate: date,
        concepts: concepts || [],
        industry: industry || '',
      });
    }

    const stats = stockMap.get(code)!;
    stats.count++;

    // 更新最新日期
    if (date > stats.latestDate) {
      stats.latestDate = date;
    }

    // 合并概念（去重）
    if (concepts && concepts.length > 0) {
      const existingConcepts = new Set(stats.concepts);
      concepts.forEach((concept) => existingConcepts.add(concept));
      stats.concepts = Array.from(existingConcepts);
    }
  });

  // 3. 转换为数组并排序（按出现次数降序）
  const statistics = Array.from(stockMap.values());
  statistics.sort((a, b) => b.count - a.count);

  return statistics;
}
```

### 趋势数据计算

```typescript
/**
 * 计算每日上榜数量趋势
 * @param dateRange 可选的日期范围
 * @returns 趋势数据数组
 */
export async function calculateTrendData(dateRange?: {
  startDate: string;
  endDate: string;
}): Promise<Array<{ date: string; count: number }>> {
  // 1. 获取记录
  const records = dateRange
    ? await getOpportunityRecordsByDateRange(dateRange.startDate, dateRange.endDate)
    : await getAllOpportunityRecords();

  // 2. 按日期分组统计
  const dateMap = new Map<string, number>();

  records.forEach((record) => {
    const date = record.date;
    dateMap.set(date, (dateMap.get(date) || 0) + 1);
  });

  // 3. 转换为数组并按日期排序
  const trendData = Array.from(dateMap.entries())
    .map(([date, count]) => ({ date, count }))
    .sort((a, b) => a.date.localeCompare(b.date));

  return trendData;
}
```

### 添加导出功能

#### 1. 实现导出函数

```typescript
// src/services/opportunity/recordService.ts
export async function exportStatisticsToCSV(statistics: StockStatistics[]): Promise<void> {
  // 1. 构建 CSV 内容
  const headers = ['股票代码', '股票名称', '出现次数', '最新日期', '所属概念', '所属行业'];
  const rows = statistics.map((stats) => [
    stats.code,
    stats.name,
    stats.count,
    stats.latestDate,
    stats.concepts.join(';'),
    stats.industry,
  ]);

  const csvContent = [headers.join(','), ...rows.map((row) => row.join(','))].join('\n');

  // 2. 创建 Blob
  const blob = new Blob(['\ufeff' + csvContent], { type: 'text/csv;charset=utf-8;' });

  // 3. 触发下载
  const link = document.createElement('a');
  const url = URL.createObjectURL(blob);
  link.setAttribute('href', url);
  link.setAttribute('download', `analysis_records_${formatDate(new Date())}.csv`);
  link.style.visibility = 'hidden';
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
}
```

## 🐛 常见问题排查

### 问题 1: 统计数据不准确

**症状**：显示的出现次数与实际不符

**排查步骤**：

1. 检查 IndexedDB 中是否有重复记录
2. 查看统计逻辑是否正确
3. 确认日期范围筛选是否生效

**解决方案**：

- 清理重复记录
- 重新计算统计数据
- 检查日期格式是否一致

### 问题 2: 趋势图不显示

**症状**：趋势图视图为空或报错

**排查步骤**：

1. 检查趋势数据是否为空
2. 查看 Ant Design Plots 是否正确安装
3. 检查数据格式是否符合要求

**解决方案**：

- 确认有分析记录
- 重新安装 `@ant-design/plots`
- 验证数据格式

### 问题 3: 概念标签显示异常

**症状**：概念标签显示为 undefined 或空

**排查步骤**：

1. 检查记录中的 concepts 字段
2. 查看渲染逻辑是否正确
3. 确认数据类型

**解决方案**：

- 确保 concepts 是数组类型
- 添加空值检查
- 提供默认值

### 问题 4: 清空操作失败

**症状**：点击清空按钮后无反应或报错

**排查步骤**：

1. 检查 IndexedDB 事务是否正常
2. 查看权限是否足够
3. 确认数据库连接

**解决方案**：

- 刷新页面重试
- 检查浏览器存储权限
- 手动清理 IndexedDB

## 📊 性能优化建议

### 1. 缓存统计结果

```typescript
const statisticsCache = new Map<string, StockStatistics[]>();

export async function getCachedStatistics(
  cacheKey: string,
  dateRange?: { startDate: string; endDate: string }
): Promise<StockStatistics[]> {
  if (statisticsCache.has(cacheKey)) {
    return statisticsCache.get(cacheKey)!;
  }

  const statistics = await calculateStockStatistics(dateRange);
  statisticsCache.set(cacheKey, statistics);

  return statistics;
}
```
