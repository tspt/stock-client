## 🏗️ 整体架构

### 数据流图

```
用户设置提醒
    ↓
保存到 localStorage
    ↓
定时轮询获取行情数据
    ↓
checkAlerts() 检查所有启用的提醒
    ↓
根据提醒类型调用相应算法
    ├── 价格/幅度 → 直接比较
    ├── 支撑阻力位 → identifySupportResistance()
    ├── 成交量异常 → detectVolumeAnomaly()
    └── 指标交叉 → detectXXXCross()
    ↓
满足条件 → 触发提醒 → 发送通知
```

## ⚙️ 核心逻辑

### 1. 支撑/阻力位识别

**文件**: `src/utils/alertEnhancements.ts`

```typescript
/**
 * 识别支撑位和阻力位
 * @param klineData K线数据
 * @param lookbackPeriod 回溯周期（默认60天）
 * @param tolerance 聚类容差（默认0.02，即2%）
 */
export function identifySupportResistance(
  klineData: KLineData[],
  lookbackPeriod: number = 60,
  tolerance: number = 0.02
): { supports: number[]; resistances: number[] } {
  // 1. 提取局部高点和低点
  const highs: number[] = [];
  const lows: number[] = [];

  for (let i = 1; i < klineData.length - 1; i++) {
    const prev = klineData[i - 1].high;
    const curr = klineData[i].high;
    const next = klineData[i + 1].high;

    if (curr > prev && curr > next) {
      highs.push(curr);
    }
  }

  // 类似逻辑提取lows...

  // 2. 价格水平聚类
  const resistances = clusterPriceLevels(highs, tolerance);
  const supports = clusterPriceLevels(lows, tolerance);

  return { supports, resistances };
}

/**
 * 价格水平聚类
 */
function clusterPriceLevels(prices: number[], tolerance: number): number[] {
  if (prices.length === 0) return [];

  const sorted = [...prices].sort((a, b) => a - b);
  const clusters: number[][] = [[sorted[0]]];

  for (let i = 1; i < sorted.length; i++) {
    const lastCluster = clusters[clusters.length - 1];
    const avg = lastCluster.reduce((sum, p) => sum + p, 0) / lastCluster.length;

    if (Math.abs(sorted[i] - avg) / avg <= tolerance) {
      lastCluster.push(sorted[i]);
    } else {
      clusters.push([sorted[i]]);
    }
  }

  // 返回每个聚类的平均值
  return clusters.map((cluster) => cluster.reduce((sum, p) => sum + p, 0) / cluster.length);
}
```

### 2. 成交量异常检测

```typescript
/**
 * 检测成交量异常
 * @param klineData K线数据
 * @param period 检测周期（默认20天）
 * @param multiplier 倍数阈值（默认2.0）
 * @param type 异常类型：spike(放量) 或 shrink(缩量)
 */
export function detectVolumeAnomaly(
  klineData: KLineData[],
  period: number = 20,
  multiplier: number = 2.0,
  type: 'spike' | 'shrink' = 'spike'
): boolean {
  if (klineData.length < period + 1) return false;

  // 计算历史平均成交量
  const historicalVolumes = klineData.slice(-period - 1, -1).map((d) => d.volume);
  const avgVolume = historicalVolumes.reduce((sum, v) => sum + v, 0) / period;

  // 当前成交量
  const currentVolume = klineData[klineData.length - 1].volume;

  if (type === 'spike') {
    return currentVolume >= avgVolume * multiplier;
  } else {
    return currentVolume <= avgVolume / multiplier;
  }
}
```

### 3. 指标金叉/死叉检测

**MACD 检测**:

```typescript
export function detectMACDCross(klineData: KLineData[]): {
  crossed: boolean;
  type: 'golden' | 'death' | null;
} {
  if (klineData.length < 2) {
    return { crossed: false, type: null };
  }

  const current = klineData[klineData.length - 1];
  const previous = klineData[klineData.length - 2];

  if (!current.macd || !previous.macd) {
    return { crossed: false, type: null };
  }

  const currentDiff = current.macd.dif - current.macd.dea;
  const previousDiff = previous.macd.dif - previous.macd.dea;

  // 金叉：DIFF从下向上穿越DEA
  if (previousDiff <= 0 && currentDiff > 0) {
    return { crossed: true, type: 'golden' };
  }

  // 死叉：DIFF从上向下穿越DEA
  if (previousDiff >= 0 && currentDiff < 0) {
    return { crossed: true, type: 'death' };
  }

  return { crossed: false, type: null };
}
```

**KDJ、RSI、MA 检测**：类似逻辑，检测快线和慢线的交叉。

### 4. 提醒检查逻辑

**文件**: `src/stores/alertStore.ts`

```typescript
async function checkAlerts(klineDataMap: Map<string, KLineData[]>) {
  const alerts = get().alerts;
  const triggeredAlerts: PriceAlert[] = [];

  for (const alert of alerts) {
    if (!alert.enabled || alert.notified) continue;

    const klineData = klineDataMap.get(alert.stockCode);
    if (!klineData || klineData.length === 0) continue;

    const currentPrice = klineData[klineData.length - 1].close;
    let shouldTrigger = false;

    switch (alert.type) {
      case AlertType.PRICE:
        if (alert.condition === AlertCondition.ABOVE) {
          shouldTrigger = currentPrice >= (alert.targetPrice || 0);
        } else {
          shouldTrigger = currentPrice <= (alert.targetPrice || 0);
        }
        break;

      case AlertType.SUPPORT_RESISTANCE:
        const { supports, resistances } = identifySupportResistance(klineData);
        if (alert.condition === AlertCondition.BREAKOUT) {
          shouldTrigger = resistances.some((r) => currentPrice >= r);
        } else {
          shouldTrigger = supports.some((s) => currentPrice <= s);
        }
        break;

      case AlertType.VOLUME_ANOMALY:
        shouldTrigger = detectVolumeAnomaly(
          klineData,
          alert.volumePeriod,
          alert.volumeMultiplier,
          alert.condition === AlertCondition.SPIKE ? 'spike' : 'shrink'
        );
        break;

      case AlertType.INDICATOR_CROSS:
        let crossResult;
        switch (alert.indicatorType) {
          case IndicatorType.MACD:
            crossResult = detectMACDCross(klineData);
            break;
          case IndicatorType.KDJ:
            crossResult = detectKDJCross(klineData);
            break;
          // ... 其他指标
        }

        if (crossResult?.crossed) {
          shouldTrigger =
            alert.condition === AlertCondition.GOLDEN_CROSS
              ? crossResult.type === 'golden'
              : crossResult.type === 'death';
        }
        break;
    }

    if (shouldTrigger) {
      triggeredAlerts.push(alert);
      sendNotification(alert, currentPrice);
    }
  }

  // 更新提醒状态
  updateAlerts(triggeredAlerts);
}
```

function checkRetracement(alert: PriceAlert, currentPrice: number) {
  if (!alert.notified) return;

  let shouldReset = false;

  if (alert.type === AlertType.PRICE) {
    if (alert.condition === AlertCondition.ABOVE) {
      shouldReset = currentPrice < alert.targetPrice! * 0.98; // 回退2%
    } else {
      shouldReset = currentPrice > alert.targetPrice! * 1.02;
    }
  }

  // 类似逻辑处理其他类型...

  if (shouldReset) {
    alert.notified = false;
  }
}

### 2. 缓存优化

```typescript
// 缓存支撑阻力位计算结果
const supportResistanceCache = new Map<
  string,
  {
    supports: number[];
    resistances: number[];
    timestamp: number;
  }
>();

function getCachedSupportResistance(stockCode: string, klineData: KLineData[]) {
  const cached = supportResistanceCache.get(stockCode);
  const now = Date.now();

  // 缓存有效期5分钟
  if (cached && now - cached.timestamp < 5 * 60 * 1000) {
    return cached;
  }

  const result = identifySupportResistance(klineData);
  supportResistanceCache.set(stockCode, {
    ...result,
    timestamp: now,
  });

  return result;
}
```
