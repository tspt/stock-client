/**
 * 检查预加载状态
 * @returns Promise<boolean> 是否已完成预加载
 */
export async function checkPreloadStatus(): Promise<boolean> {
  // 可以通过检查localStorage或特定标记来判断
  const lastPreloadTime = localStorage.getItem('constituent_stocks_preload_time');
  if (!lastPreloadTime) return false;

  const preloadTimestamp = parseInt(lastPreloadTime);
  const now = Date.now();
  const oneDay = 24 * 60 * 60 * 1000;

  // 如果预加载时间在24小时内，认为有效
  return now - preloadTimestamp < oneDay;
}

/**
 * 标记预加载完成（由用户确认后调用）
 */
export function markPreloadCompleted(): void {
  localStorage.setItem('constituent_stocks_preload_time', Date.now().toString());
}

### 2. 数据服务层

**文件**: `src/services/stocks/constituent-stocks.ts`

**核心函数**:

```typescript
import { getIndustrySectorStocks } from '../hot/industry-sectors';
import { getConceptSectorStocks } from '../hot/concept-sectors';
import { getAllIndustrySectors } from '../hot/industry-sectors';
import { getAllConceptSectors } from '../hot/concept-sectors';

/**
 * 获取所有成分股数据
 * @param category 板块类型: 'industry' | 'concept'
 * @param signal AbortSignal 用于取消请求
 */
export async function getAllConstituentStocks(
  category: 'industry' | 'concept' = 'industry',
  signal?: AbortSignal
): Promise<ConstituentStockData[]> {
  try {
    let allStocks: ConstituentStockData[] = [];

    if (category === 'industry') {
      // 获取所有行业板块
      const industries = await getAllIndustrySectors();

      // 并行获取每个行业的成分股
      const stockPromises = industries.map(async (industry) => {
        try {
          const result = await getIndustrySectorStocks(
            industry.code,
            'f3', // 按涨跌幅排序
            1, // 降序
            100, // 每页100条（最大支持）
            1, // 第1页
            signal
          );

          return result.data.map((stock) => ({
            code: stock.code,
            name: stock.name,
            sectorType: 'industry' as const,
            sectorCode: industry.code,
            sectorName: industry.name,
            price: stock.price || 0,
            changePercent: stock.changePercent || 0,
            volume: 0, // 需要根据实际API调整
            turnover: 0, // 需要根据实际API调整
          }));
        } catch (error) {
          console.warn(`获取行业 ${industry.name} 成分股失败:`, error);
          return [];
        }
      });

      const stocksArrays = await Promise.all(stockPromises);
      allStocks = stocksArrays.flat();
    } else {
      // 获取所有概念板块
      const concepts = await getAllConceptSectors();

      // 并行获取每个概念的成份股
      const stockPromises = concepts.map(async (concept) => {
        try {
          const result = await getConceptSectorStocks(
            concept.code,
            'f3', // 按涨跌幅排序
            1, // 降序
            100, // 每页100条（最大支持）
            1, // 第1页
            signal
          );

          return result.data.map((stock) => ({
            code: stock.code,
            name: stock.name,
            sectorType: 'concept' as const,
            sectorCode: concept.code,
            sectorName: concept.name,
            price: stock.price || 0,
            changePercent: stock.changePercent || 0,
            volume: 0, // 需要根据实际API调整
            turnover: 0, // 需要根据实际API调整
          }));
        } catch (error) {
          console.warn(`获取概念 ${concept.name} 成分股失败:`, error);
          return [];
        }
      });

      const stocksArrays = await Promise.all(stockPromises);
      allStocks = stocksArrays.flat();
    }

    return allStocks;
  } catch (error) {
    console.error('获取成分股数据失败:', error);
    throw error;
  }
}
```

### 3. 数据导出功能

**文件**: `src/utils/export/constituent-stocks-export.ts`

```typescript
import type { ConstituentStockData } from '@/types/stock';

/**
 * 导出成分股数据为CSV格式
 * @param stocks 成分股数据数组
 * @param filename 文件名
 */
export function exportConstituentStocksToCSV(
  stocks: ConstituentStockData[],
  filename: string = 'constituent_stocks.csv'
): void {
  // CSV头部
  const headers = [
    '股票代码',
    '股票名称',
    '板块类型',
    '板块代码',
    '板块名称',
    '最新价',
    '涨跌幅(%)',
    '成交量',
    '成交额',
  ];

  // 构建CSV内容
  const csvContent = [
    headers.join(','),
    ...stocks.map((stock) =>
      [
        stock.code,
        stock.name,
        stock.sectorType === 'industry' ? '行业' : '概念',
        stock.sectorCode,
        stock.sectorName,
        stock.price.toFixed(2),
        stock.changePercent.toFixed(2),
        stock.volume,
        stock.turnover,
      ].join(',')
    ),
  ].join('\n');

  // 创建下载链接
  const blob = new Blob(['\ufeff' + csvContent], { type: 'text/csv;charset=utf-8;' });
  const link = document.createElement('a');
  const url = URL.createObjectURL(blob);

  link.setAttribute('href', url);
  link.setAttribute('download', filename);
  link.style.visibility = 'hidden';

  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
}

/**
 * 导出成分股数据为JSON格式
 * @param stocks 成分股数据数组
 * @param filename 文件名
 */
export function exportConstituentStocksToJSON(
  stocks: ConstituentStockData[],
  filename: string = 'constituent_stocks.json'
): void {
  const jsonContent = JSON.stringify(stocks, null, 2);
  const blob = new Blob([jsonContent], { type: 'application/json;charset=utf-8;' });
  const link = document.createElement('a');
  const url = URL.createObjectURL(blob);

  link.setAttribute('href', url);
  link.setAttribute('download', filename);
  link.style.visibility = 'hidden';

  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
}
```

### 4. 自动刷新机制

```typescript
// ConstituentStocksPage.tsx
useEffect(() => {
  // 首次加载
  loadData();

  // 设置定时刷新（每5分钟）
  const timer = setInterval(() => {
    loadData(true); // true表示静默刷新
  }, 5 * 60 * 1000);

  return () => clearInterval(timer);
}, [category]);

async function loadData(silent = false) {
  if (!silent) setLoading(true);
  try {
    const result = await getAllConstituentStocks(category);
    setData(result);
  } catch (error) {
    console.error('Failed to load constituent stocks:', error);
  } finally {
    if (!silent) setLoading(false);
  }
}
```

### 3. 数据缓存策略

```typescript
// 简单的内存缓存
const cache = new Map<string, { data: any; timestamp: number }>();
const CACHE_TTL = 5 * 60 * 1000; // 5分钟缓存

export async function getCachedConstituentStocks(
  category: 'industry' | 'concept'
): Promise<ConstituentStockData[]> {
  const cacheKey = `constituent_stocks_${category}`;
  const cached = cache.get(cacheKey);

  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.data;
  }

  const data = await getAllConstituentStocks(category);
  cache.set(cacheKey, { data, timestamp: Date.now() });

  return data;
}
```
