```typescript
// 获取所有股票历史数据
export async function getStocksHistory(codes: string[]): Promise<StockHistory[]> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onsuccess = (event) => {
      const db = (event.target as IDBOpenDBRequest).result;
      const transaction = db.transaction(['stocks_history'], 'readonly');
      const store = transaction.objectStore('stocks_history');

      if (codes.length === 0) {
        // 获取所有数据
        const getAllRequest = store.getAll();
        getAllRequest.onsuccess = () => resolve(getAllRequest.result);
      } else {
        // 获取指定股票
        const results: StockHistory[] = [];
        let completed = 0;

        codes.forEach((code) => {
          const getRequest = store.get(code);
          getRequest.onsuccess = () => {
            if (getRequest.result) {
              results.push(getRequest.result);
            }
            completed++;
            if (completed === codes.length) {
              resolve(results);
            }
          };
        });
      }
    };
  });
}

// 保存回测结果
export async function batchSaveSignalBacktests(results: SignalBacktest[]): Promise<void> {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onsuccess = (event) => {
      const db = (event.target as IDBOpenDBRequest).result;
      const transaction = db.transaction(['signal_backtests'], 'readwrite');
      const store = transaction.objectStore('signal_backtests');

      results.forEach((result) => {
        store.put(result);
      });

      transaction.oncomplete = () => resolve();
      transaction.onerror = () => reject(transaction.error);
    };
  });
}
```

```typescript
async function handleBacktestStock(payload: {
  code: string;
  name: string;
  klineData: KLineData[];
  timeRange: number;
}) {
  const { code, name, klineData, timeRange } = payload;

  try {
    // 1. 识别交易信号
    const signals = detectSignals(klineData, timeRange);

    // 2. 验证每个信号
    const signalResults: SignalResult[] = [];
    for (const signal of signals) {
      const result = await validateSignal(signal, klineData);
      signalResults.push(result);
    }

    // 3. 发送结果回主线程
    self.postMessage({
      type: 'BACKTEST_RESULT',
      payload: {
        code,
        name,
        signals: signalResults,
      },
    });
  } catch (error) {
    self.postMessage({
      type: 'BACKTEST_ERROR',
      payload: { code, error: error.message },
    });
  }
}
```

```typescript
function detectSignals(klineData: KLineData[], timeRange: number): Signal[] {
  const signals: Signal[] = [];

  // 遍历K线数据，识别买点信号
  for (let i = 0; i < klineData.length; i++) {
    const currentDay = klineData[i];

    // 使用ML模型判断是否为买点
    const isBuyPoint = mlBuypointModel.predict(currentDay, klineData.slice(0, i));

    if (isBuyPoint) {
      signals.push({
        date: currentDay.date,
        price: currentDay.close,
        index: i,
      });
    }
  }

  return signals;
}
```

### 信号验证

```typescript
async function validateSignal(signal: Signal, klineData: KLineData[]): Promise<SignalResult> {
  const signalIndex = signal.index;
  const returns: number[] = [];

  // 计算后续1日、2日、3日的收益率
  for (let days = 1; days <= 3; days++) {
    const futureIndex = signalIndex + days;

    if (futureIndex < klineData.length) {
      const entryPrice = signal.price;
      const exitPrice = klineData[futureIndex].close;
      const returnRate = ((exitPrice - entryPrice) / entryPrice) * 100;
      returns.push(returnRate);
    } else {
      returns.push(0); // 数据不足
    }
  }

  return {
    signalDate: signal.date,
    entryPrice: signal.price,
    returns,
  };
}
```

```typescript
import * as tf from '@tensorflow/tfjs';

let model: tf.LayersModel | null = null;
let modelMetadata: ModelMetadata | null = null;

// 加载模型
export async function loadModel(): Promise<void> {
  if (model) return; // 已加载

  try {
    // 从本地文件加载模型
    model = await tf.loadLayersModel('/models/buypoint-model-v4/model.json');
    modelMetadata = await fetchModelMetadata();

    logger.info('ML买点模型加载成功');
  } catch (error) {
    logger.error('ML买点模型加载失败:', error);
    throw error;
  }
}

// 预测买点
export function predictBuyPoint(currentDay: KLineData, historyData: KLineData[]): boolean {
  if (!model) {
    throw new Error('模型未加载');
  }

  // 1. 特征工程
  const features = extractFeatures(currentDay, historyData);

  // 2. 转换为Tensor
  const inputTensor = tf.tensor2d([features]);

  // 3. 预测
  const prediction = model.predict(inputTensor) as tf.Tensor;
  const probability = prediction.dataSync()[0];

  // 4. 清理内存
  inputTensor.dispose();
  prediction.dispose();

  // 5. 判断是否为买点（置信度 > 0.7）
  return probability > 0.7;
}

// 获取模型元数据
export function getModelMetadata(): ModelMetadata {
  return modelMetadata || DEFAULT_METADATA;
}
```

### 特征工程

```typescript
function extractFeatures(currentDay: KLineData, historyData: KLineData[]): number[] {
  const features: number[] = [];

  // 1. 价格特征
  features.push(currentDay.close);
  features.push(currentDay.open);
  features.push(currentDay.high);
  features.push(currentDay.low);

  // 2. 技术指标
  const ma5 = calculateMA(historyData, 5);
  const ma10 = calculateMA(historyData, 10);
  const ma20 = calculateMA(historyData, 20);

  features.push(ma5);
  features.push(ma10);
  features.push(ma20);

  // 3. 成交量特征
  features.push(currentDay.volume);
  const avgVolume = calculateAvgVolume(historyData, 5);
  features.push(currentDay.volume / avgVolume);

  // 4. 涨跌幅
  if (historyData.length > 0) {
    const prevClose = historyData[historyData.length - 1].close;
    const changePercent = ((currentDay.close - prevClose) / prevClose) * 100;
    features.push(changePercent);
  }

  // 归一化
  return normalizeFeatures(features);
}
```

```typescript
// 将信号按股票分组
function groupSignalsByStock(signals: any[]): any[] {
  const grouped: Map<string, any> = new Map();

  signals.forEach((signal) => {
    const { code, name, signalDate, entryPrice, returns } = signal;

    if (!grouped.has(code)) {
      grouped.set(code, {
        code,
        name,
        signals: [],
        totalSignals: 0,
        avgReturn: 0,
        winRate: 0,
      });
    }

    const group = grouped.get(code)!;
    group.signals.push({ signalDate, entryPrice, returns });
    group.totalSignals++;
  });

  // 计算统计信息
  grouped.forEach((group) => {
    const allReturns = group.signals.flatMap((s) => s.returns);
    group.avgReturn = allReturns.reduce((sum, r) => sum + r, 0) / allReturns.length;

    const winCount = allReturns.filter((r) => r > 0).length;
    group.winRate = (winCount / allReturns.length) * 100;
  });

  return Array.from(grouped.values());
}
```

```typescript
// src/workers/backtestWorker.ts

// 均线交叉策略
function movingAverageCrossStrategy(
  klineData: KLineData[],
  shortPeriod: number = 5,
  longPeriod: number = 20
): Signal[] {
  const signals: Signal[] = [];

  for (let i = longPeriod; i < klineData.length; i++) {
    const shortMA = calculateMA(klineData.slice(i - shortPeriod, i), shortPeriod);
    const longMA = calculateMA(klineData.slice(i - longPeriod, i), longPeriod);

    const prevShortMA = calculateMA(klineData.slice(i - shortPeriod - 1, i - 1), shortPeriod);
    const prevLongMA = calculateMA(klineData.slice(i - longPeriod - 1, i - 1), longPeriod);

    // 金叉：短期均线上穿长期均线
    if (prevShortMA <= prevLongMA && shortMA > longMA) {
      signals.push({
        date: klineData[i].date,
        price: klineData[i].close,
        index: i,
        type: 'MA_CROSS_GOLDEN',
      });
    }
  }

  return signals;
}
```

## 🐛 常见问题排查

### 问题 1: 回测速度很慢

**症状**：回测大量股票时耗时过长

**排查步骤**：

1. 检查 CPU 使用率
2. 查看 Worker 是否正常工作
3. 检查是否有内存泄漏

**解决方案**：

- 减少单次回测的股票数量
- 增加批次大小（batchSize）
- 优化特征提取算法
- 使用更高效的浏览器（Chrome/Edge）

### 问题 2: Worker 通信失败

**症状**：回测结果无法返回到主线程

**排查步骤**：

1. 打开浏览器控制台
2. 查看是否有 Worker 错误
3. 检查消息格式是否正确

**解决方案**：

- 确保消息格式符合约定
- 检查数据类型是否可序列化
- 避免传递过大的对象

### 问题 3: ML 模型加载失败

**症状**：提示"模型未加载"

**排查步骤**：

1. 检查模型文件是否存在
2. 查看网络请求是否成功
3. 检查 TensorFlow.js 版本兼容性

**解决方案**：

- 确认模型文件路径正确
- 重新下载模型文件
- 更新 TensorFlow.js 版本

### 问题 4: IndexedDB 存储空间不足

**症状**：保存数据时失败

**排查步骤**：

1. 检查浏览器存储限制
2. 查看 IndexedDB 使用情况
3. 清理过期数据

**解决方案**：

- 定期清理旧的回测结果
- 压缩存储的数据
- 使用分页加载
