### 1. 数据服务层

```typescript
// 获取行业板块列表
export async function getIndustrySectors(
  sortType: string = 'f3',
  sortOrder: number = 1,
  pageSize: number = 50,
  pageNum: number = 1
): Promise<{ data: IndustrySectorRankData[]; total: number }> {
  const response = await fetch('/api/eastmoney/clist/get', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      fs: 'm:90+s:4', // 行业板块标识
      fid: sortType, // 排序字段
      po: sortOrder.toString(), // 排序方向
      pz: pageSize.toString(), // 每页数量
      pn: pageNum.toString(), // 页码
      fields: 'f12,f14,f2,f3,f62,f184,f66,f69,f72,f75,f78,f81,f84,f87,f204,f205,f124,f1,f13',
    }),
  });

  const text = await response.text();
  const jsonMatch = text.match(/\((.*)\)/);
  if (!jsonMatch || !jsonMatch[1]) {
    throw new Error('无法解析JSONP响应');
  }

  const rawData = JSON.parse(jsonMatch[1]);
  return parseIndustrySectorData(rawData.data.diff, rawData.data.total);
}
```

### 2. 数据解析

```typescript
function parseIndustrySectorData(
  rawList: any[],
  total: number
): { data: IndustrySectorRankData[]; total: number } {
  const data: IndustrySectorRankData[] = rawList.map((item) => ({
    code: item.f12,
    name: item.f14,
    changePercent: item.f3,

    // 主力净流入（元转万元）
    mainNetInflow: item.f62 ? item.f62 / 10000 : undefined,
    mainNetInflowRatio: item.f184,

    // 超大单净流入（元转万元）
    superLargeNetInflow: item.f66 ? item.f66 / 10000 : undefined,
    superLargeNetInflowRatio: item.f69,

    // 大单净流入（元转万元）
    largeNetInflow: item.f72 ? item.f72 / 10000 : undefined,
    largeNetInflowRatio: item.f75,

    // 中单净流入（元转万元）
    mediumNetInflow: item.f78 ? item.f78 / 10000 : undefined,
    mediumNetInflowRatio: item.f81,

    // 小单净流入（元转万元）
    smallNetInflow: item.f84 ? item.f84 / 10000 : undefined,
    smallNetInflowRatio: item.f87,

    // 领涨股
    leadingStock: item.f204,
    leadingStockCode: item.f205,
  }));

  return { data, total };
}
```

### 3. 自动刷新机制

```typescript
// IndustrySectorPage.tsx
useEffect(() => {
  // 首次加载
  loadData();

  // 设置10秒轮询
  const timer = setInterval(() => {
    loadData(true); // true表示静默刷新
  }, 10000);

  return () => clearInterval(timer);
}, [currentPage, sortOrder]);

async function loadData(silent = false) {
  if (!silent) setLoading(true);
  try {
    const result = await getIndustrySectors('f3', sortOrder, 50, currentPage);
    setData(result.data);
    setTotal(result.total);
  } catch (error) {
    console.error('Failed to load industry sectors:', error);
  } finally {
    if (!silent) setLoading(false);
  }
}
```

### 2. 金额格式化

```typescript
const formatAmount = (value?: number): string => {
  if (!value) return '-';
  const absValue = Math.abs(value);
  if (absValue >= 10000) {
    // 大于等于1亿，显示为亿
    const yiValue = value / 10000;
    return `${yiValue.toFixed(2)}亿`;
  }
  // 否则显示为万元
  return `${value.toFixed(2)}万`;
};
```

### 2. 请求参数详解

**列表查询参数**:

```javascript
{
  fs: 'm:90+s:4',              // 行业板块标识
  fid: 'f3',                   // 排序字段：f3=涨跌幅
  po: '1',                     // 排序方向：1=降序, 0=升序
  pz: '50',                    // 每页数量
  pn: '1',                     // 页码
  fields: 'f12,f14,f2,f3,f62,f184,f66,f69,f72,f75,f78,f81,f84,f87,f204,f205,f124,f1,f13'
}
```

**数据字段映射**:

| 字段代码 | 含义             | 说明                    |
| -------- | ---------------- | ----------------------- |
| f12      | 板块代码         | 如 BK1020               |
| f14      | 板块名称         | 如 "非金属材料 Ⅱ"       |
| f3       | 涨跌幅           | 百分比                  |
| f62      | 主力净流入净额   | 单位:元(前端转换为万元) |
| f184     | 主力净流入净占比 | 百分比                  |
| f66/f69  | 超大单净额/占比  | 同上                    |
| f72/f75  | 大单净额/占比    | 同上                    |
| f78/f81  | 中单净额/占比    | 同上                    |
| f84/f87  | 小单净额/占比    | 同上                    |
| f204     | 领涨股名称       | -                       |
| f205     | 领涨股代码       | -                       |
